function updateHUD(d){ if(!d)return; const fmt=(v)=>'₹'+(Number(v||0)/100000).toFixed(2)+'L'; document.getElementById('hudAvailable').textContent=fmt(d.available); const dayEl=document.getElementById('hudDayPnL'); dayEl.textContent=(d.day_pnl>=0?'+':'')+fmt(Math.abs(d.day_pnl)); dayEl.style.color=d.day_pnl>=0?'var(--tds-success)':'var(--tds-danger)'; const totEl=document.getElementById('hudTotalPnL'); totEl.textContent=(d.total_pnl>=0?'+':'')+fmt(Math.abs(d.total_pnl)); totEl.style.color=d.total_pnl>=0?'var(--tds-success)':'var(--tds-danger)'; }
